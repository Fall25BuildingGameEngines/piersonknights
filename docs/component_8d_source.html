234 string data = readText(filename);
235 JSONValue animData = parseJSON(data);
237 auto formatData = animData["format"];
238
239 int width = formatData["width"].integer.to!int;
240 int height = formatData["height"].integer.to!int;

242 int tileHeight = formatData["tileHeight"].integer.to!int;
243
245 int rows = height/ tileHeight;
246 int cols = width/ tileWidth;
247
249 mFrames.length = rows * cols;
250
251 int frameIndex = 0;
253 for (int y = 0; y < rows; ++y)
254 {
256 for (int x = 0; x < cols; ++x)
257 {
259 SDL_FRect rect;
260 rect.x = x * tileWidth;
261 rect.y = y * tileHeight;
262 rect.w = tileWidth;
263 rect.h = tileHeight;
265 Frame f;
266 f.mRect = rect;
267 f.mElapsedTime = 0;
268 mFrames[frameIndex++] = f;
269 }
270 }
272 auto allAnims = animData["frames"].object;
274 foreach (animName, frameArray; allAnims)
275 {
276 long[] indices;
278 foreach (f; frameArray.array)
279 indices ~= f.integer;
280
282 mFrameNumbers[animName] = indices;
283 }
