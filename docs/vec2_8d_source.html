202// Reflect a vector across a surface normal.
203// 'v' is the incoming (incident) vector
204// 'n' is the surface normal of 'v'
205// *Hint* Normalizing the vectors makes this easier
206// Helpful chapter: https://immersivemath.com/ila/ch03_dotproduct/ch03.html
207// Another pragmatic resource: https://www.sunshine2k.de/articles/coding/vectorreflection/vectorreflection.html
208// I used this formula for my implementation at the top:
209// https://www.contemporarycalculus.com/dh/Calculus_all/CC11_7_VectorReflections.pdf
210T Reflect(T)(T v, T n){
211 T result;
212
213 auto len = sqrt(cast(double)(n.x * n.x + n.y * n.y));
214 if (len == 0) {
215 result.x = 0;
216 result.y = 0;
217 return result;
218 }
219
220 auto nx = n.x / len;
221 auto ny = n.y / len;
222
223 // dot product
224 auto dot = v.x * nx + v.y * ny;
225
226 // get reflection
227 result.x = v.x - 2 * dot * nx;
228 result.y = v.y - 2 * dot * ny;
229
230 return result;
231}

233// Projects vector 'u' onto vector 'v' and returns the
234// new vector.
235T Project(T)(T u, T v){
236 T result;
237
238 auto dotUV = Dot(u, v);
239 auto dotVV = Dot(v,v);
240
241 if (dotVV == 0) {
242 result.x = 0;
243 result.y = 0;
244 return result;
245 }
246
247 auto scale = dotUV / dotVV;
248
249 result.x = v.x * scale;
250 result.y = v.y * scale;
251
252 return result;
253}
